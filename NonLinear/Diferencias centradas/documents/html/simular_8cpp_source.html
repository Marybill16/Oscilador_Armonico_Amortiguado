57/**
58 * @brief Calcula la velocidad y el momentum usando diferencias centradas.
59 *
60 * Para los puntos internos se aplica la fórmula de derivación centrada,
61 * mientras que en los extremos la velocidad se asigna como cero, ya que
62 * no es posible calcularla sin datos anteriores y posteriores.
63 * Luego se calcula el momentum multiplicando la velocidad por la masa.
64 *
65 * @param tiempo Arreglo con los valores de tiempo.
66 * @param posicion Arreglo con los valores de posición.
67 * @param velocidad Arreglo donde se almacenarán las velocidades calculadas.
68 * @param momentum Arreglo donde se almacenará el momentum calculado.
69 * @param n Número total de datos.
70 */
71void calcularDerivadas(const double tiempo[], const double posicion[], double velocidad[], double momentum[], int n) {
72 for (int i = 0; i < n; ++i) {
73 if (i > 0 && i < n - 1) { // Derivada centrada
74 velocidad[i] = (posicion[i + 1] - posicion[i - 1]) / (tiempo[i + 1] - tiempo[i - 1]);
75 } else {
76 velocidad[i] = 0.0; // No se puede calcular para los extremos
77 }
78 }
79 for (int i = 0; i < n; ++i){
80 momentum[i] = velocidad[i] * m;
81 }
82}
