54/**
55 * @brief Calcula las derivadas del sistema de ecuaciones diferenciales.
56 *
57 * Evalúa las ecuaciones del oscilador amortiguado:
58 * - dx/dt = v
59 * - dv/dt = -(k/m)*x - (l/m)*v*|v|
60 *
61 * @param dxdt Referencia para almacenar la derivada de la posición (dx/dt).
62 * @param dvdt Referencia para almacenar la derivada de la velocidad (dv/dt).
63 */
64void funcion(double x, double t, double v, double m, double k, double l, double &dxdt, double &dvdt) {
65 dxdt = v; ///< Derivada de la posición: velocidad.
66 dvdt = (-k/m) * x - (l/m) * v * fabs(v); ///< Derivada de la velocidad: aceleración (fuerza elástica + amortiguamiento cuadrático).
67}

69/**
70 * @brief Aplica un paso del método de Runge-Kutta de cuarto orden.
71 *
72 * Avanza la solución del sistema de ecuaciones diferenciales un paso de tiempo dt,
73 * actualizando los valores de posición (x) y velocidad (v).
74 *
75 */
76void rungeKutta4(double &x, double &t, double &v, double &m, double &k, double &l, double dt) {
77 double dx1, dv1, dx2, dv2, dx3, dv3, dx4, dv4; ///< Variables para las evaluaciones intermedias de Runge-Kutta.
78
79 funcion(x, t, v, m , k, l, dx1, dv1); ///< Primera evaluación (k1).
80 funcion(x + 0.5 * dt * dx1, t + 0.5 * dt, v + 0.5 * dt * dv1, m , k, l, dx2, dv2); ///< Segunda evaluación (k2).
81 funcion(x + 0.5 * dt * dx2, t + 0.5 * dt, v + 0.5 * dt * dv2, m , k, l, dx3, dv3); ///< Tercera evaluación (k3).
82 funcion(x + dt * dx3, t + dt, v + dt * dv3, m , k, l, dx4, dv4); ///< Cuarta evaluación (k4).
83
84 /// Actualización de x y v usando la combinación ponderada de las evaluaciones.
85 x += dt * (dx1 + 2 * dx2 + 2 * dx3 + dx4) / 6.0;
86 v += dt * (dv1 + 2 * dv2 + 2 * dv3 + dv4) / 6.0;
87}

89/**
90 * @brief Ejecuta la simulación y guarda los resultados en un archivo.
91 *
92 * Realiza la integración numérica desde t0 hasta tmax con paso dt,
93 * guardando en un archivo los valores de tiempo, posición, velocidad y momento lineal.
94 * Al finalizar, ejecuta un script de gnuplot para graficar los resultados.
95 *
96 * @param archivo Ruta del archivo de salida donde se guardarán los datos.
97 * @param x0 Posición inicial (m).
98 * @param t0 Tiempo inicial (s).
99 * @param v0 Velocidad inicial (m/s).
100 * @param tmax Tiempo máximo de simulación (s).
101 */
102void guardarDatos(const string &archivo, double x0, double t0, double v0, double m, double k,
103 double l, double dt, double tmax) {
104 system("mkdir -p ../results"); ///< Crea el directorio de resultados si no existe.
105
106 ofstream file(archivo);
107 file << fixed << setprecision(6); ///< Formato de salida: punto fijo con 6 decimales.
108 file << setw(10) << "t" << setw(15) << "x" << setw(15) << "v" << setw(15) << "p" <<endl; ///< Encabezado del archivo.
109
110 double x = x0; ///< Posición inicial.
111 double v = v0; ///< Velocidad inicial.
112 double p = m * v; ///< Momento lineal inicial (p = m*v).
113
114 /// Bucle de integración temporal.
115 for (double t = t0; t <= tmax; t += dt) {
116 file << setw(10) << t << setw(15) << x << setw(15) << v << setw(15) << p << endl; ///< Escribe datos en el archivo.
117 rungeKutta4(x, t, v, m, k ,l, dt); ///< Avanza un paso de tiempo.
118 p = m * v; ///< Actualiza el momento lineal.
119 }
120
121 file.close(); ///< Cierra el archivo de salida.
122 cout << "Datos guardados en: " << archivo << endl;
123
124 system ("gnuplot ../scripts/graficar.gp"); ///< Ejecuta el script de gnuplot para generar gráficas.
125}
